separate(date, sep = "/", into = c("month", "day", "year"), convert = TRUE ) %>%
  mutate(
    month = month.name[as.numeric(month)]) %>%
  mutate(
    year, year = if_else (year <= 15, year + 2000, year + 1900)) %>%
